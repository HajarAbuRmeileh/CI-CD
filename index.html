import math c=int(input()) d=int(input()) def add_numbers(c,d): return c+d def abstract_numbers(c,d): return c-d def mult_numbers(c,d): return c*d def power_numbers(c,d): return math.pow(c,d) print("First number X : ",c) print("Second number Y: ",d) print("--------------------------------------------------------") sum=add_numbers(c,d) pow=power_numbers(c,d) abstract=abstract_numbers(c,d) mult=mult_numbers(c,d) print("X+Y Result =",sum) print("X-Y Result =",abstract) print("X*Y Result =",mult) print("X^Y Result =",pow)
